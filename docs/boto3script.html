The following script was written to iterate through all of CloudHesive's managed customer accounts and list all ec2 instances with specified tags and data. The diagram above summarizes the process.

  1. Python function runs from CloudHesive's production account.
  2. Function reads DynamoDB table "awsaccountroles" and retrieves ARN.
  3. Function generates temporary STS token from ARN.
  4. Using generated STS token, function assumes role and gathers data of all ec2 instances in every region of account.

import pandas as pd
import boto3
from boto3.dynamodb.conditions import Attr

boto3.setup_default_session(profile_name='default')

#Initiate dynamodb client, access awsaccountroles table, return items from dynamodb table with role
dynamodb_client = boto3.resource('dynamodb')
aws_account_table = dynamodb_client.Table('awsaccountroles')
response = aws_account_table.scan(FilterExpression=Attr('role').ne('null'))

#Convert response into dataframe, select relevent columns from response_df to create df
response_df = pd.DataFrame(response['Items'])
df = response_df[['id','fullymanaged','name','role','org']]

#Select fullymanaged US Production accounts, create ARN and store in ARN column
df = df.loc[(df['fullymanaged'] == 'true') & (df['org'] == 'CloudHesive US Production')]
df['arn'] = df.apply(lambda row: 'arn:aws:iam::' + row.id + ':role/' + row.role, axis=1)

#Generate master_empty_list to throw data in. Generate credentials_list to store temp credentials.
master_empty_list = []
credentials_list = []

#Create iter_list to iterate through ARN's, create temporary credentials, start session and gather data.
iter_list = df[['name', 'arn']].to_dict('records')

#Creating temporary credentials and storing to credentials_list
for i in iter_list:
    sts_client = boto3.client('sts')
    role_object = sts_client.assume_role(RoleArn=i['arn'],
                                         RoleSessionName=i['name'],
                                         DurationSeconds=900)
    keyid = role_object['Credentials']['AccessKeyId']
    key = role_object['Credentials']['SecretAccessKey']
    token = role_object['Credentials']['SessionToken']
    credentials_list.append([i['name'], keyid, key, token])   
    
#Use temporary credentials to start session, obtain regions, iterate through regions and get ec2 data
for x in credentials_list:
    session = boto3.Session(aws_access_key_id=x[1], aws_secret_access_key=x[2], aws_session_token=x[3])
    ec2_client = session.client('ec2')

    regions = ec2_client.describe_regions()
    
    for q in regions['Regions']:
        ec2 = session.client('ec2', region_name = q['RegionName'])
        instances = ec2.describe_instances()
        if instances['Reservations'] != []:
            master_empty_list.append([x[0], q['RegionName'], instances])
            print(x[0], q['RegionName'])

#Initiate dataframe list
df_list = []

#Define extractor that will extract data from master_empty_list, format it into dataframe
def extractor(master_list):
    company = master_list[0]
    region = master_list[1]
    instance_dict = master_list[2]
    for a in instance_dict['Reservations']:
        for b in a['Instances']:
            instance_id = b['InstanceId']
            instance_type = b['InstanceType']
            launch_date = str(b['LaunchTime'])
            tags = b['Tags']
            if 'Platform' in b.keys():
                platform = b['Platform']
            else:
                platform = 'no_platform'
            if 'IamInstanceProfile' in b.keys():
                instance_profile_arn = b['IamInstanceProfile']['Arn']
                instance_profile_id = b['IamInstanceProfile']['Id']
            else:
                instance_profile_arn = 'no_instance_profile'
                instance_profile_id = 'no_instance_profile'
           
            df_list.append([company, region, instance_id, instance_type, 
                            launch_date, platform, instance_profile_arn, 
                            instance_profile_id, tags])

#Use extractor function and create master_df (master dataframe)
for w in master_empty_list:
    extractor(w)
    
master_df = pd.DataFrame(df_list, columns = ['company', 'region', 'instanceid', 'type', 'launchdate',
                                             'platform', 'instanceprofile_arn', 'instanceprofile_id', 'tags'])

#Define tag column extractor to extract data from tag column, tag_extractor is then applied via lambda function
def tag_extractor(row, tag):
    tag_set = ""
    for v in row:
        if v['Key'] == tag:
            tag_set = v['Value']
            return tag_set

#Define lambda applier that runs lambda function across each row in dataframe and returns master_df
def lambda_applier(column_name, column):
    master_df[column_name] = master_df.apply(lambda row: tag_extractor(row.tags, column), axis = 1)
    return master_df

#Create list of tags to analyze and run functions on
taglist1 = ['cloudhesivemanaged', 'env', 'role', 'managedby', 'prioritylevel', 'ownedby', 'securitylevel', 
            'provisionedby', 'commissiondate', 'supportedby', 'source', 'steadystate']

for g in taglist1:
    column_tag = g + '_tag'
    lambda_applier(column_tag, g)

#Drop tags column
master_df.drop(columns = ['tags'], inplace = True)
